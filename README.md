3. npm install
4. create a dotenv file

JWT_SECRET=3d7e5f8c2b1a9d6e4f2c8b7a3d1e9f5c2b4a8d6e3f7c9b1a5e2d8f4c6b3a7d9e
PORT=5000
EMAIL_FROM=your_email@example.com
6. npm start

7. go to the frontend directory
8. cd frontend
9. npm install

    SKIP_PREFLIGHT_CHECK=true
